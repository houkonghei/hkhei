小鎮 (Node) 是否確定最短路路径？ 預估最短時間 (分鐘) 前一步從小鎮 (Parent)

💻 其實很簡單！C++ 程式碼對應實作 (C++98)

程式碼中使用了直觀的二維陣列 (Adjacency Matrix) 與手動尋找最小值，不需使用複雜的樣板容器，最適合初學者理解！

dijkstra_algorithm.cpp C++98 經典版 (使用二維陣列) 
#include <iostream>

using namespace std;

// 定義代表無限大的數值（代表一開始路還不通，設為十億）
const int INF = 1000000000; 

// 定義地圖上最大的城鎮數量
const int MAX_N = 6;

void dijkstra(int start, int graph[MAX_N][MAX_N], int n) {
    // 1. 建立我們在網頁中看到的那張「記錄表」
    int dist[MAX_N];       // 記錄從起點到各小鎮的最短時間，一開始都設為無限大
    bool visited[MAX_N];   // 記錄小鎮是否已被「標記為確定」，一開始都設為 false (未確定)
    int parent[MAX_N];     // 記錄各小鎮的最短路徑是從「哪一個前一步小鎮」走過來的，一開始都設為 -1

    // 2. 初始化資料表 (C++98 寫法：用簡單的 for 迴圈逐一設定初始值)
    for (int i = 0; i < n; ++i) {
        dist[i] = INF;
        visited[i] = false;
        parent[i] = -1;
    }

    // 郵局 (起點) 到自己的時間是 0 分鐘
    dist[start] = 0;

    // 3. 開始執行 Dijkstra 演算法的主迴圈 (每次確定一個小鎮，共跑 n 次)
    for (int step = 0; step < n; ++step) {
        
        // 【第一步】從「還沒確定」的小鎮中，手動找出目前預估時間最短的那一個
        int u = -1;
        int min_val = INF;
        
        for (int i = 0; i < n; ++i) {
            if (!visited[i] && dist[i] < min_val) {
                min_val = dist[i];
                u = i;
            }
        }

        // 如果找不到任何可到達的小鎮，代表剩下的城鎮都不連通，可以提前結束
        if (u == -1) {
            break;
        }

        // 【第二步】將此小鎮標記為「已確定最短路徑」
        visited[u] = true;

        // 【第三步】檢查這個小鎮連出去的所有鄰近小鎮 (Relaxation 鬆弛操作)
        for (int v = 0; v < n; ++v) {
            // 如果 u 小鎮與 v 小鎮之間有路 (權重不為 INF)，且 v 小鎮還沒有被確定
            if (graph[u][v] != INF && !visited[v]) {
                // 如果：【走到 u 的時間】 + 【u 到 v 的道路時間】 < 【原本記錄走到 v 的時間】
                if (dist[u] + graph[u][v] < dist[v]) {
                    dist[v] = dist[u] + graph[u][v]; // 更新成更短、更省時的時間！
                    parent[v] = u;                  // 記錄前一步是從小鎮 u 走過來的
                }
            }
        }
    }

    // 4. 輸出最終送信的最短時間結果
    cout << "--- Dijkstra 送信路線表結果 ---" << endl;
    for (int i = 0; i < n; ++i) {
        char town = 'A' + i; // 轉換為字母 A, B, C...
        cout << "到小鎮 " << town << " 的最短時間: ";
        if (dist[i] == INF) {
            cout << "路不通 (INF)" << endl;
        } else {
            cout << dist[i] << " 分鐘" << endl;
        }
    }
}

int main() {
    int n = 6; // 共 6 個小鎮 (A=0, B=1, C=2, D=3, E=4, F=5)
    
    // 建立一個 6x6 的二維陣列 (Adjacency Matrix)，代表地圖上的所有小鎮與道路
    int graph[MAX_N][MAX_N];

    // 初始化地圖：一開始假設所有小鎮之間都沒有路 (設為無限大 INF)
    // 自己到自己則設為 0 分鐘
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (i == j) {
                graph[i][j] = 0;
            } else {
                graph[i][j] = INF;
            }
        }
    }

    // 手動填入地圖上的道路與行車時間 (因為是雙向通車，所以對稱的兩個位置都要設定)
    graph[0][1] = graph[1][0] = 4;  // A - B (4 分鐘)
    graph[0][2] = graph[2][0] = 2;  // A - C (2 分鐘)
    graph[1][2] = graph[2][1] = 1;  // B - C (1 分鐘)
    graph[1][3] = graph[3][1] = 5;  // B - D (5 分鐘)
    graph[2][3] = graph[3][2] = 8;  // C - D (8 分鐘)
    graph[2][4] = graph[4][2] = 10; // C - E (10 分鐘)
    graph[3][4] = graph[4][3] = 2;  // D - E (2 分鐘)
    graph[3][5] = graph[5][3] = 6;  // D - F (6 分鐘)
    graph[4][5] = graph[5][4] = 3;  // E - F (3 分鐘)

    // 從小鎮 A (0) 開始送信！
    dijkstra(0, graph, n);

    return 0;
}
