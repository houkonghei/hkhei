小鎮 (Node) 是否確定最短路徑？ 預估最短時間 (分鐘) 前一步從小鎮 (Parent)

💻 其實很簡單！C++ 程式碼對應實作 (C++11)

程式碼中使用了優先佇列 (Priority Queue) 來加速尋找下一個最近小鎮，這正是導航系統的寫法喔！

dijkstra_algorithm.cpp C++11 Ready 
#include <iostream>
#include <vector>
#include <queue>
#include <utility>

using namespace std;

// 定義代表無限大的數值（代表一開始路還不通）
const int INF = 1e9; 

// 代表道路的結構：(目標小鎮, 送信所需時間)
struct Edge {
    int to;
    int weight;
};

void dijkstra(int start, const vector<vector<Edge>>& graph, int n) {
    // 1. 建立我們在網頁中看到的那張「記錄表」
    vector<int> dist(n, INF);     // 記錄從起點到各小鎮的最短時間，一開始都設為無限大
    vector<bool> visited(n, false); // 記錄小鎮是否已被「標記為確定」
    vector<int> parent(n, -1);     // 記錄各小鎮的最短路徑是從「哪一個前一步小鎮」走過來的

    // 2. 郵局 (起點) 到自己的時間是 0 分鐘
    dist[start] = 0;

    // 優先佇列 (Min-Heap)：用來每次自動挑選「目前時間最短、且還沒確定」的小鎮
    // 儲存格式為: pair<當前距離, 小鎮編號>
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        // 3. 挑選目前預估時間最短的小鎮
        int u = pq.top().second;
        pq.pop();

        // 如果這個小鎮已經確定過了，就直接跳過
        if (visited[u]) continue;

        // 4. 將其標記為「已確定最短路徑」
        visited[u] = true;

        // 5. 檢查這個小鎮連出去的所有鄰近小鎮 (Relaxation 鬆弛操作)
        for (const auto& edge : graph[u]) {
            int v = edge.to;
            int weight = edge.weight;

            // 如果：【走到 u 的時間】 + 【u 到 v 的時間】 < 【原本記錄走到 v 的時間】
            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight; // 更新成更短的時間！
                parent[v] = u;              // 記錄前一步是從小鎮 u 走過來的
                pq.push({dist[v], v});      // 丟進排程，之後繼續往外探索
            }
        }
    }

    // 6. 輸出最終送信的最短時間結果
    cout << "--- Dijkstra 送信路線表結果 ---" << endl;
    for (int i = 0; i < n; ++i) {
        char town = 'A' + i; // 轉換為字母 A, B, C...
        cout << "到小鎮 " << town << " 的最短時間: ";
        if (dist[i] == INF) cout << "路不通 (INF)" << endl;
        else cout << dist[i] << " 分鐘" << endl;
    }
}

int main() {
    int n = 6; // 共 6 個小鎮 (A=0, B=1, C=2, D=3, E=4, F=5)
    vector<vector<Edge>> graph(n);

    // 建立地圖上的道路與行車時間 (無向圖，雙向可通)
    // A (0) 連接的小路
    graph[0].push_back({1, 4});  // A - B (4)
    graph[0].push_back({2, 2});  // A - C (2)
    graph[1].push_back({0, 4});
    graph[2].push_back({0, 2});

    // B (1) 連接的小路
    graph[1].push_back({2, 1});  // B - C (1)
    graph[1].push_back({3, 5});  // B - D (5)
    graph[2].push_back({1, 1});
    graph[3].push_back({1, 5});

    // C (2) 連接的小路
    graph[2].push_back({3, 8});  // C - D (8)
    graph[2].push_back({4, 10}); // C - E (10)
    graph[3].push_back({2, 8});
    graph[4].push_back({2, 10});

    // D (3) 連接的小路
    graph[3].push_back({4, 2});  // D - E (2)
    graph[3].push_back({5, 6});  // D - F (6)
    graph[4].push_back({3, 2});
    graph[5].push_back({3, 6});

    // E (4) 連接的小路
    graph[4].push_back({5, 3});  // E - F (3)
    graph[5].push_back({4, 3});

    // 從小鎮 A (0) 開始送信！
    dijkstra(0, graph, n);

    return 0;
}
